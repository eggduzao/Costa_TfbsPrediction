\begin{multicols}{2} % \begin{multicols}{#columns}[header text][space]
\begin{footnotesize} % scriptsize(7) < footnotesize(8) < small (9) < normal (10)
\nomenclature{bp}{Base Pair (pares de bases); representa um par de bases (nucleotídeos) no DNA, isto é, uma coordenada genômica.}

\nomenclature{SNP}{Single Nucleotide Polymorphism (polimorfismos de único nucleotídeo); variações pontuais (em apenas um nucleotídeo) no genoma.}

\nomenclature{TF}{Transcription Factor (fator de transcrição); elementos regulatórios trans-atuantes. São proteínas que se ligam em regiões específicas no genoma para regular a transcrição de um ou mais genes.}

\nomenclature{TFBS}{Transcription Factor Binding Site (sítio de ligação de fatores de transcrição); elementos regulatórios cis-atuantes. São as regiões onde os fatores de transcrição se ligam.}

\nomenclature{Motif}{padrão frequente ou assinatura; sequência genômica ou proteômica com padrão reconhecível e que tenha significado biológico.}

\nomenclature{MM}{Motif Matching; técnica computacional que utiliza representações probabilísticas de \emph{motifs} (PFMs, PSSMs ou PWMs) para atribuir um grau de afinidade para regiões genômicas a respeito da probabilidade de um fator de transcrição se ligar àquela região}

\nomenclature{DNase~I}{Desoxirribonuclease I; endonuclease codificada pelo gene \emph{DNASE1} capaz de clivar o DNA em várias diferentes condições.}

\nomenclature{Endonuclease}{Classe de proteínas que clivam as ligações fosfodiéster dentro de uma cadeia de DNA.}

\nomenclature{ChIP}{Chromatin Immunoprecipitation (imunoprecipitação da cromatina); técnica biológica para recuperar regiões genômicas onde uma proteína de interesse está ligada, através da imunoprecipitação da mesma utilizando um anticorpo (ou outros materiais).}

\nomenclature{ChIP-chip}{Chromatin Immunoprecipitation followed by chip (imunoprecipitação da cromatina seguida de chip); técnica biológica para identificar regiões genômicas onde uma proteína de interesse está ligada através da realização de ChIP seguida de experimentos com \emph{tiling arrays}.}

\nomenclature{ChIP-seq}{Chromatin Immunoprecipitation followed by massive sequencing (imunoprecipitação da cromatina seguida de sequenciamento massivo); técnica biológica para identificar regiões genômicas onde uma proteína de interesse está ligada através da realização de ChIP seguida de sequenciamento massivo dos fragmentos genômicos recuperados.}

\nomenclature{DNase-chip}{DNase~I digestion followed by chip (digestão por DNase~I seguida de chip); técnica biológica para identificação de regiões de cromatina descondensada através da clivagem do DNA com a endonuclease DNase~I seguida de experimentos com \emph{tiling arrays}.}

\nomenclature{DNase-seq}{DNase~I digestion followed by massive sequencing (digestão por DNase~I seguida de sequenciamento massivo); técnica biológica para identificação de regiões de cromatina descondensada através da clivagem do DNA com a endonuclease DNase~I seguida de sequenciamento massivo dos fragmentos genômicos recuperados.}

\nomenclature{Microarray}{Microarranjo; técnica experimental para medir níveis de expressão gênica (ou alguns outros atributos) que utiliza um chip que contém diversos fragmentos de DNA que representam regiões de interesse (genes ou exons, por exemplo).}

\nomenclature{Tiling array}{técnica experimental semelhante ao microarranjo, porém neste caso os fragmentos de DNA no chip representam regiões contíguas no genoma dada uma janela e deslocamento específico.}

\nomenclature{Acetilação}{Reação que introduz um grupo funcional acetila em um composto orgânico.}

\nomenclature{Metilação}{Reação que introduz um grupo funcional metila em um composto orgânico.}

\nomenclature{Fosforilação}{Reação que introduz um grupo funcional fosfato em um composto orgânico.}

\nomenclature{Ubiquitinação}{Marcação através de moléculas ubiquitina.}

\nomenclature{HMM}{Hidden Markov Model (modelo escondido de Markov); técnica para modelagem estatística de séries temporais baseada em processos estocásticos de Markov.}

\nomenclature{ATP}{Adenosina Trifosfato; nucleotídeo responsável pelo armazenamento de energia em suas ligações químicas, utilizado em reações que exigem tal energia.}

\nomenclature{LCR}{Locus Control Regions; região composta por vários elementos cis-atuantes distais cuja composição representa a sua funcionalidade regulatória.}

\nomenclature{GTF}{General Transcription Factors (fatores de transcrição gerais); conjunto de proteínas que, junto com a RNA polimerase e o mediador, constituem o aparato básico para que a transcrição ocorra em nível basal em eucariotos.}

\nomenclature{PIC}{Transcription Preinitiation Complex (complexo pré-iniciação de transcrição); complexo de proteínas montados na região promotora necessárias para a transcrição.}

\nomenclature{TSS}{Transcription Start Site (sítios de início de transcrição); sítio onde a transcrição se inicia.}

\nomenclature{Inr}{Elemento Iniciador; elemento presente no núcleo do promotor de alguns genes relacionado com a formação do complexo pré-iniciação.}

\nomenclature{DCE}{Downstream Core Elements; elemento presente no núcleo do promotor de alguns genes relacionado com a formação do complexo pré-iniciação.}

\nomenclature{BRE}{TFIIB-Recognition Element; elemento presente no núcleo do promotor de alguns genes relacionado com a formação do complexo pré-iniciação.}

\nomenclature{MTE}{Motif Ten Element; elemento presente no núcleo do promotor de alguns genes relacionado com a formação do complexo pré-iniciação.}

\nomenclature{EDTA}{Ethylenediamine Tetraacetic Acid (ácido etilenodiamino tetra-acético); composto orgânico que age como agente quelante, formando complexos muito estáveis com diversos íons metálicos. Das várias utilizações destaca-se o controle em experimentos de ChIP.}

\nomenclature{PCR}{Polymerase Chain Reaction (reação em cadeia da polimerase); método de amplificação (de criação de múltiplas cópias) de DNA.}

\nomenclature{PFM}{Position Frequency Matrix (matriz de frequência de posição); representação matricial de um \emph{motif} onde as linhas representam os nucleotídeos e as colunas representam as posições do \emph{motif}.}

\nomenclature{PSSM}{Position Specific Scoring Matrix (matrizes de pontuação específica por posição); neste trabalho está sendo utilizado como sinônimo de PWM.}

\nomenclature{PWM}{Position Weight Matrix (matrizes de peso de posição); representação matricial logarítmica de um \emph{motif} criada através de uma PFM.}

\nomenclature{MPSS}{Massively Parallel Signature Sequencing; abordagem utilizada para identificar e quantificar transcritos de mRNA presentes em uma amostra.}

\nomenclature{ENCODE}{Encyclopedia of DNA Elements; Iniciativa dentro do programa \emph{Genome Browser} da Universidade da Califórnia em Santa Cruz que disponibiliza diversas faixas de dados relativos à genômica funcional.}

\nomenclature{FAIRE}{Formaldehyde-Assisted Identification of Regulatory Elements; técnica biológica para identificação de regiões de cromatina descondensada através de um protocolo menos denso do que o do DNase-seq.}

\nomenclature{HS}{DNase~I Hypersensitive Sites (sítios hipersensíveis à DNase~I); regiões no DNA que permitem a clivagem através da endonuclease DNase~I.}

\nomenclature{MPBS}{Motif Predicted Binding Sites (sítios de ligação preditos através de \emph{motifs}); termo utilizado para referenciar sítios de ligação de fatores de transcrição preditos através de \emph{motif matching}.}

\nomenclature{GHMM}{General Hidden Markov Model Library; Biblioteca disponível em C e em Python que implementa de forma eficiente HMMs com emissões discretas ou contínuas.}

\nomenclature{Python}{Linguagem de programação de alto nível, interpretada, imperativa, orientada a objetos, de tipagem dinâmica e forte. Utilizada para analisar os dados, aplicar os métodos e gerar os gráficos em todo o projeto.}

\nomenclature{FMR1}{Fragile X Mental Retardation 1; gene responsável pela codificação da proteína FMRP, comumente encontrada no cérebro e essencial para o desenvolvimento cognitivo e reprodução em fêmeas.}

\nomenclature{STAMP}{Ferramenta computacional utilizada para, entre outras funcionalidades, encontrar evidências de \emph{motifs} de fatores de transcrição em pequenos fragmentos de DNA a partir de repositórios inteiros de PWMs.}

\nomenclature{MACS}{Model-based Analysis for ChIP-Seq; Ferramenta utilizada para analisar (processar e encontrar picos) dados de ChIP-seq.}

\nomenclature{Biopython}{Conjunto de bibliotecas para a linguagem Python contendo implementações de diversas ferramentas biológicas necessárias em várias áreas de bioinformática.}

\nomenclature{EM}{Expectation Maximization (maximização da esperança); Algoritmo iterativo com objetivo de encontrar a estimativa de parâmetros de máxima verossimilhança utilizando dados sem rótulos (isto é, não se sabe a classe dos padrões).}

\nomenclature{dNTP}{desoxirribonucleotídeo trifosfato; monômero do DNA em seu formato com três grupos fosfato, necessários para produzir a energia suficiente para a interação com a macromolécula de DNA.}

\nomenclature{rNTP}{ribonucleotídeo trifosfato; monômero do RNA em seu formato com três grupos fosfato, necessários para produzir a energia suficiente para a interação com a macromolécula de RNA.}
